void BubbleSort(int v[5])
{
 for(int k = 5 ; k > 1 ; k--)
 {
   for(int l = 1 ; l < k ; l++)
   {
      if(a[l-1] > a[l]){
         swap(&a[l-1],&a[l]);
      }
   }
 }
}
 
void swap(int* v1, int* v2)
{
 int temp = *v1;
 *v1 = *v2;
 *v2 = temp;
}
